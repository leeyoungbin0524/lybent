const productId = new URLSearchParams(location.search).get("product_id"); async function requireLogin(){ const { data:{ user }, error } = await sb.auth.getUser(); if(error || !user){ location.href = `login.html?next=${encodeURIComponent("checkout.html" + location.search)}`; return null; } return user; } // ✅ 주소 검색 function openDaumPostcode(){ new daum.Postcode({ oncomplete: function(data){ const postalEl = document.getElementById("postal"); const addr1El = document.getElementById("addr1"); const addr2El = document.getElementById("addr2"); const extraEl = document.getElementById("addrExtra"); if(postalEl) postalEl.value = data.zonecode || ""; const baseAddr = data.roadAddress || data.jibunAddress || ""; if(addr1El) addr1El.value = baseAddr; // 참고항목 표시(법정동/건물명) let extra = ""; if (data.bname && /[동|로|가]$/g.test(data.bname)) extra += data.bname; if (data.buildingName && data.apartment === "Y") extra += (extra ? ", " : "") + data.buildingName; extra = extra ? `참고: (${extra})` : ""; if(extraEl) extraEl.textContent = extra; // 상세주소 포커스 if(addr2El) addr2El.focus(); } }).open(); } let product = null; async function loadProduct(){ const box = document.getElementById("productBox"); if(!productId){ box.innerHTML = `
❌ 상품 정보가 없습니다.
`; return; } const { data, error } = await sb .from("products") .select("id,name,price,stock,is_active,image_url,description") .eq("id", productId) .maybeSingle(); if(error || !data){ box.innerHTML = `
❌ 상품을 불러올 수 없음
${(error?.message||"")}
`; return; } product = data; box.innerHTML = `
${product.image_url ? `` : ``}
${product.name}
가격: ${product.price ?? "—"} / 재고: ${product.stock ?? "—"}
${product.description ?? ""}
`; } document.getElementById("checkoutForm").addEventListener("submit", async (e)=>{ e.preventDefault(); setMsg("주문 생성 중…"); const user = await requireLogin(); if(!user) return; if(!product){ setMsg("❌ 상품 로딩 실패"); return; } // ✅ 주소 필수(원하면 풀어도 됨) const postal = (document.getElementById("postal").value || "").trim(); const addr1 = (document.getElementById("addr1").value || "").trim(); if(!postal || !addr1){ setMsg("❌ 주소 검색으로 배송지를 입력해 주세요."); return; } const qty = Math.max(1, parseInt(document.getElementById("qty").value || "1", 10)); const payload = { user_id: user.id, status: "pending", product_id: product.id, product_name: product.name, price: product.price ?? null, qty, buyer_name: (document.getElementById("buyerName").value || "").trim(), buyer_phone: (document.getElementById("buyerPhone").value || "").trim(), buyer_email: user.email || null, shipping_postal: postal, shipping_address1: addr1, shipping_address2: (document.getElementById("addr2").value || "").trim(), memo: (document.getElementById("memo").value || "").trim(), }; const { error } = await sb.from("orders").insert(payload); if(error){ setMsg("❌ 주문 생성 실패: " + error.message); return; } setMsg("✅ 주문이 생성됐습니다! (결제는 준비중)"); }); window.addEventListener("DOMContentLoaded", async ()=>{ // 주소검색 버튼 연결 const btn = document.getElementById("addrSearchBtn"); if(btn) btn.addEventListener("click", openDaumPostcode); await loadProduct(); });
